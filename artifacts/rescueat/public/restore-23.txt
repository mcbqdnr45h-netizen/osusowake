OK=0

echo "--- ダウンロード開始 ---"

  mkdir -p "$(dirname "$f")"
  if curl -fsS "$BASE/$f" -o "$f.dl"; then
    if [ -s "$f.dl" ]; then
      mv "$f.dl" "$f"
      OK=$((OK+1))
      printf "."
    else
      rm -f "$f.dl"
      echo ""
      echo "FAIL (empty): $f"
      FAIL=$((FAIL+1))
    fi

    rm -f "$f.dl"
    echo ""
